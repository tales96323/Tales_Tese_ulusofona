    pdftitle={\thesistitle},
    pdfauthor={\thesisauthor},
    pdfsubject={\thesisdegree},
    pdfkeywords={Requirements Engineering; PageRank; HITS; SALSA; AST; Graph Ranking},

% ============================================================================
%  BACK MATTER (Glossário, Apêndices, Anexos)
% ============================================================================
% Glossário: capítulo não numerado do back matter.
\include{back/glossario}
